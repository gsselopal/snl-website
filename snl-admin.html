Database Setup

Run this SQL once in your Supabase SQL editor to create the required tables. Go to Supabase Dashboard → SQL Editor → New Query and paste the code below.

-- SNL Website Schema
-- Run this in Supabase SQL Editor

create table if not exists news_items (
  id          bigint generated always as identity primary key,
  title       text not null,
  description text,
  category    text default 'general',
  date        text,
  link        text,
  link_label  text,
  created_at  timestamptz default now()
);

create table if not exists team_members (
  id          bigint generated always as identity primary key,
  name        text not null,
  status      text default 'msc',
  role        text,
  focus       text,
  badge       text,
  email       text,
  created_at  timestamptz default now()
);

create table if not exists journal_covers (
  id          bigint generated always as identity primary key,
  title       text not null,
  journal     text,
  year        text,
  img_url     text not null,
  link        text,
  created_at  timestamptz default now()
);

-- Enable Row Level Security (read-only for public)
alter table news_items    enable row level security;
alter table team_members  enable row level security;
alter table journal_covers enable row level security;

-- Allow public read access
create policy "Public read news"    on news_items    for select using (true);
create policy "Public read team"    on team_members  for select using (true);
create policy "Public read covers"  on journal_covers for select using (true);

-- Allow anon full access (for admin portal) — tighten in production
create policy "Anon write news"    on news_items    for all using (true) with check (true);
create policy "Anon write team"    on team_members  for all using (true) with check (true);
create policy "Anon write covers"  on journal_covers for all using (true) with check (true);
